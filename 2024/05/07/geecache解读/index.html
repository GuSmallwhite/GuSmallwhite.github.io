Day1 LRU缓存淘汰

这一天主要是实现LRU缓存淘汰策略

数据结构与构造函数

type Cache struct {
	maxBytes int64//允许使用的最大内存
	nbytes   int64//已经使用的内存
	ll       *list.List//双向链表，用于lru中的双向链表
	cache    map[string]*list.Element//LRU中的HashMap，用于key和键值对的映射
	// optional and executed when an entry is purged.
	OnEvicted func(key string, value Value)//回调函数
}

键值对

type entry struct {
	key   string
	value Value
}

key为string，value是自定义类型

Value

// Value use Len to count how many bytes it takes
type Value interface {
	Len() int//返回使用内存的大小
}

构造函数

// New is the Constructor of Cache
func New(maxBytes int64, onEvicted func(string, Value)) *Cache {
	return &Cache{
		maxBytes:  maxBytes,
		ll:        list.New(),
		cache:     make(map[string]*list.Element),
		OnEvicted: onEvicted,
	}
}

LRU功能函数

查找函数

// Get look ups a key's value
func (c *Cache) Get(key string) (value Value, ok bool) {
	if ele, ok := c.cache[key]; ok {//如果找到了key对应的ele
		c.ll.MoveToFront(ele)//移动到队头
		kv := ele.Value.(*entry)//强制转换为entry类型
		return kv.value, true
	}
	return//找不到直接返回
}

缓存淘汰函数

// RemoveOldest removes the oldest item
func (c *Cache) RemoveOldest() {
	ele := c.ll.Back()//找到队尾
	if ele != nil {
		c.ll.Remove(ele)
		kv := ele.Value.(*entry)
		delete(c.cache, kv.key)
		c.nbytes -= int64(len(kv.key)) + int64(kv.value.Len())//更新当前所用的内存
		if c.OnEvicted != nil {
			c.OnEvicted(kv.key, kv.value)//回调函数
		}
	}
}

新增/修改

// Add adds a value to the cache.
func (c *Cache) Add(key string, value Value) {
	if ele, ok := c.cache[key]; ok {//已经存在了那就是修改
		c.ll.MoveToFront(ele)
		kv := ele.Value.(*entry)//还是强转
		c.nbytes += int64(value.Len()) - int64(kv.value.Len())
		kv.value = value
	} else {
		ele := c.ll.PushFront(&entry{key, value})//没有就新增一个
		c.cache[key] = ele
		c.nbytes += int64(len(key)) + int64(value.Len())
	}
	for c.maxBytes != 0 && c.maxBytes < c.nbytes {//如果超过最大内存，不断删除最晚的
		c.RemoveOldest()
	}
}

sync.Mutex类

就是一个锁，用法

var m sync.Mutex
var set = make(map[int]bool, 0)

func printOnce(num int) {
	m.Lock()//上锁
	if _, exist := set[num]; !exist {
		fmt.Println(num)
	}
	set[num] = true
	m.Unlock()//解锁
}

当然你也可以lock后立马defer

func printOnce(num int) {
	m.Lock()
	defer m.Unlock()
	if _, exist := set[num]; !exist {
		fmt.Println(num)
	}
	set[num] = true
}

ByteView

一个数据结构，用于表示缓存的值

ByteView实现了上篇定义的Value接口

在go中，一个类只要实现了一个接口的所有方法，那么就认为他实现了这个接口

// A ByteView holds an immutable view of bytes.
type ByteView struct {
	b []byte
}

// return view's length
func (v ByteView) Len() int {
	return len(v.b)
}

// return a copy of the data as a byte slice,so the value can't be modified
func (v ByteView) ByteSlice() []byte {
	return cloneBytes(v.b)
}

// returns the data as a string
func (v ByteView) String() string {
	return string(v.b)
}

//获取值的时候返回一个拷贝
func cloneBytes(b []byte) []byte {
	c := make([]byte, len(b))
	copy(c, b)
	return c
}

cache类

包装了lru类和锁，用于实现并发

type cache struct {
	mu         sync.Mutex
	lru        *lru.Cache//包装了lru数据结构
	cacheBytes int64
}

//在方法调用前后添加lock，unlock
func (c *cache) add(key string, value ByteView) {
	c.mu.Lock()
	defer c.mu.Unlock()
	if c.lru == nil {
		c.lru = lru.New(c.cacheBytes, nil)
	}
	c.lru.Add(key, value)
}

func (c *cache) get(key string) (value ByteView, ok bool) {
	c.mu.Lock()
	defer c.mu.Unlock()
	if c.lru == nil {
		return
	}

	if v, ok := c.lru.Get(key); ok {
		return v.(ByteView), ok
	}

	return
}

Group数据结构初见

Group 是 GeeCache 最核心的数据结构，负责与用户的交互，并且控制缓存值存储和获取的流程。

代码框架

geecache/
    |--lru/
        |--lru.go  // lru 缓存淘汰策略
    |--byteview.go // 缓存值的抽象与封装
    |--cache.go    // 并发控制
    |--geecache.go // 负责与外部交互，控制缓存存储和获取的主流程

在这里的流程：

                         是
接收 key --> 检查是否被缓存 -----> 返回缓存值 ⑴            
                            |  否
                            |-----> 调用`回调函数`，获取值并添加到缓存 --> 返回缓存值 ⑶

回调函数Getter

一个空函数，让用户自定义如果没有缓存，那么从哪里获取

// A Getter loads data for a key.
type Getter interface {
	Get(key string) ([]byte, error)
}

// A GetterFunc implements Getter with a function.
type GetterFunc func(key string) ([]byte, error)

// Get implements Getter interface function
func (f GetterFunc) Get(key string) ([]byte, error) {
	return f(key)
}
  • 定义接口 Getter 和 回调函数 Get(key string)([]byte, error)，参数是 key，返回值是 []byte。
  • 定义函数类型 GetterFunc，并实现 Getter 接口的 Get 方法。
  • 函数类型实现某一个接口，称之为接口型函数，方便使用者在调用时既能够传入函数作为参数，也能够传入实现了该接口的结构体作为参数。

Group的定义

包装了回调函数和cache数据结构，用于控制流程

以及自己的名字，所以就是缓存分类吧，name缓存， class缓存等等类似的

type Group struct {
	name      string
	getter    Getter
	mainCache cache
}

var (
	mu     sync.RWMutex
	groups = make(map[string]*Group)//构造name和group的对应关系
)

// NewGroup creates a new instance of Groupt
func NewGroup(name string, cacheBytes int64, getter Getter) *Group {
	if getter == nil {
		panic("nil Getter")
	}
	mu.Lock()
	defer mu.Unlock()
	g := &Group{
		name:      name,
		getter:    getter,
		mainCache: cache{cacheBytes: cacheBytes},
	}
	groups[name] = g//创建一个group放到groups里
	return g
}

// GetGroup returns the named group previously created with NewGroup, or
// nil if there's no such group.
func GetGroup(name string) *Group {
	mu.RLock()
	g := groups[name]
	mu.RUnlock()
	return g
}

Get方法

实现了上面所说的流程，见注释

func (g *Group) Get(key string) (ByteView, error) {
	if key == " " {
		return ByteView{}, nil
	}
	if v, ok := g.mainCache.get(key); ok {
		log.Println("[Geecache]hit")
		return v, nil//找到了直接返回
	}
	return g.load(key)//没找到load
}

func (g *Group) load(key string) (value ByteView, err error) {
	return g.getLocally(key)//在这里的流程是流程3，从本地获取
}

func (g *Group) getLocally(key string) (ByteView, error) {
	bytes, err := g.getter.Get(key)//之前定义的回调函数，用户自定义从哪个地方获取
	if err != nil {
		return ByteView{}, err
	}
	value := ByteView{b: cloneBytes(bytes)}
	g.populateCache(key, value)
	return value, nil
}
func (g *Group) populateCache(key string, value ByteView) {
	g.mainCache.add(key, value)//这个就是直接push进去就行了
}

http.ListenAndServe 接收 2 个参数，第一个参数是服务启动的地址，第二个参数是 Handler，任何实现了 ServeHTTP 方法的对象都可以作为 HTTP 的 Handler。

方法

实现ServeHTTP方法，作为HTTP的Handler

// Log info with server name
func (p *HTTPPool) Log(format string, v ...interface{}) {
	log.Printf("[Server %s] %s", p.self, fmt.Sprintf(format, v...))
}

//根据请求当中的路径，从groupmap当中获取对应的group，key对应的数据
func (p *HTTPPool) ServeHTTP(w http.ResponseWriter, r *http.Request) {
	//url should be started with basePath
	if !strings.HasPrefix(r.URL.Path, p.basePath) {
		panic("HTTPPool serving unexpected path: " + r.URL.Path)
	}
	p.Log("%s %s", r.Method, r.URL.Path)
	//分割路径
	parts := strings.SplitN(r.URL.Path[len(p.basePath):], "/", 2)
	if len(parts) != 2 {
		http.Error(w, "bad request", http.StatusBadRequest)
		return
	}
	//提取group名和ke
	groupName := parts[0]
	key := parts[1]

	group := GetGroup(groupName)//获取group
	if group == nil {
		http.Error(w, "no such group: "+groupName, http.StatusNotFound)
		return
	}
	view, err := group.Get(key)//获取value
	if err != nil {
		http.Error(w, err.Error(), http.StatusInternalServerError)
		return
	}
	w.Header().Set("Content-Type", "application/octet-stream")
	w.Write(view.ByteSlice())//以responsebody返回
}

DAY4一致性哈希

这里的一致性哈希的用途是什么？

使用一致性哈希，使得每一个key对应相同的节点

什么意思，我有groupmap对不对，groupmap存储了groupname和group的对应关系

group又存储了key和value的对应关系

这里就用一致性哈希，使得我不同的key映射到相同的peer上，并且在节点增加删除后，变动较少

这里的比较简单，主要就是一致性哈希的实现，没有涉及到后面和节点以及HTTP客户端结合的部分

属实比较孤立了

来看代码就行了

// Hash maps bytes to uint32
type Hash func(data []byte) uint32

// Map constains all hashed keys
type Map struct {
	hash     Hash//哈希算法
	replicas int//每一个虚拟节点的个数
	keys     []int // 逻辑环
	hashMap  map[int]string//虚拟节点的哈希值和实际节点名称的对应关系
}

// New creates a Map instance
func New(replicas int, fn Hash) *Map {
	m := &Map{
		replicas: replicas,
		hash:     fn,
		hashMap:  make(map[int]string),
	}
	if m.hash == nil {
		m.hash = crc32.ChecksumIEEE
	}
	return m
}

添加和获取，都比较简单

// Add adds some keys to the hash.
func (m *Map) Add(keys ...string) {
	for _, key := range keys {//为每一个key，构造replicas个数的虚拟节点
		for i := 0; i < m.replicas; i++ {
			hash := int(m.hash([]byte(strconv.Itoa(i) + key)))
			m.keys = append(m.keys, hash)
			m.hashMap[hash] = key
		}
	}
	sort.Ints(m.keys)
}
//Get gets the closest item in the hash to the provided key.
func (m *Map) Get(key string) string {
	if len(m.keys) == 0 {
		return ""
	}

	hash := int(m.hash([]byte(key)))
	// Binary search for appropriate replica.
	idx := sort.Search(len(m.keys), func(i int) bool {
		return m.keys[i] >= hash
	})

	return m.hashMap[m.keys[idx%len(m.keys)]]
}
  • 选择节点就非常简单了，第一步，计算 key 的哈希值。
  • 第二步，顺时针找到第一个匹配的虚拟节点的下标 idx，从 m.keys 中获取到对应的哈希值。如果 idx == len(m.keys)，说明应选择 m.keys[0]，因为 m.keys 是一个环状结构，所以用取余数的方式来处理这种情况。
  • 第三步，通过 hashMap 映射得到真实的节点。

PeerPicker

有一说一一直没懂这个peer是什么玩意

// PeerPicker is the interface that must be implemented to locate
// the peer that owns a specific key.
type PeerPicker interface {
	PickPeer(key string) (peer PeerGetter, ok bool)
    
}

// PeerGetter is the interface that must be implemented by a peer.
type PeerGetter interface {
	Get(group string, key string) ([]byte, error)
}

目测来开，PeerPicker 的 PickPeer() 方法用于根据传入的 key 选择相应节点 PeerGetter。

然后接口 PeerGetter 的 Get() 方法用于从对应 group 查找缓存值。PeerGetter 就对应于上述流程中的 HTTP 客户端。

HTTP客户端

之前写了服务端，就是解析请求然后调用group.get然后把内容写进response里

这里要写客户端，什么意思呢，就是要和远程节点的服务端通信

type httpGetter struct {
	baseURL string//要访问的远程url地址，比如http://example.com/_geecache/
}

//这一步对应从远程节点获取
//httpGetter实现了PeerGetter接口
func (h *httpGetter) Get(group string, key string) ([]byte, error) {
	u := fmt.Sprintf(
		"%v%v/%v",
		h.baseURL,
		url.QueryEscape(group),//防止特殊字符
		url.QueryEscape(key),
	)
	res, err := http.Get(u)//发送get请求
	if err != nil {
		return nil, err
	}
	defer res.Body.Close()

	if res.StatusCode != http.StatusOK {
		return nil, fmt.Errorf("server returned: %v", res.Status)
	}

	bytes, err := ioutil.ReadAll(res.Body)//读取响应，保存在bytes[]当中
	if err != nil {
		return nil, fmt.Errorf("reading response body: %v", err)
	}

	return bytes, nil
}

实现节点选择

// HTTPPool implements PeerPicker for a pool of HTTP peers.
type HTTPPool struct {
	// this peer's base URL, e.g. "https://example.net:8000"
	self        string
	basePath    string
	mu          sync.Mutex // guards peers and httpGetters
	peers       *consistenthash.Map
	httpGetters map[string]*httpGetter // keyed by e.g. "http://10.0.0.2:8008"
}

  • 新增成员变量 peers，类型是一致性哈希算法的 Map，用来根据具体的 key 选择节点。

  • 新增成员变量 httpGetters，映射远程节点与对应的 httpGetter。每一个远程节点对应一个 httpGetter，因为 httpGetter 与远程节点的地址 baseURL 有关。

还记得一致性哈希算法的 Map是什么样的吗？

// Map constains all hashed keys
type Map struct {
	hash     Hash
	replicas int
	keys     []int // 逻辑环
	hashMap  map[int]string//保存哈希值和真实节点名的对应关系
}

所以把这个Map放到HTTPPool当中，就把一致性哈希和HTTP结合在一起了

实现PeerPicker接口

// Set updates the pool's list of peers.
func (p *HTTPPool) Set(peers ...string) {//添加传入的节点
	p.mu.Lock()
	defer p.mu.Unlock()
	p.peers = consistenthash.New(defaultReplicas, nil)
	p.peers.Add(peers...)
	p.httpGetters = make(map[string]*httpGetter, len(peers))
	for _, peer := range peers {
		p.httpGetters[peer] = &httpGetter{baseURL: peer + p.basePath}//每个节点对应一个HTTPGetter
	}
}

// HTTPPool实现了PeerPicker接口
func (p *HTTPPool) PickPeer(key string) (PeerGetter, bool) {
	p.mu.Lock()
	defer p.mu.Unlock()
	if peer := p.peers.Get(key);/*包装一致性哈希的get方法，获取节点对应的HTTPGetter*/ peer != "" && peer != p.self {
		p.Log("Pick peer %s", peer)
		return p.httpGetters[peer], true
	}
	return nil, false
}
